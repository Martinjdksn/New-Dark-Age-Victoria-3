### Character triggers
has_eldritch_trait =
{
	OR = { 
		has_trait = nda_cultist 
		has_trait = nda_acolyte 
		has_trait = nda_master    
	}
}

has_yigian_trait =
{
	OR = { 
		has_trait = nda_cultist 
		has_trait = nda_acolyte 
		has_trait = nda_master    
	}
}

has_any_cultist_trait = {
	OR = { 
		has_trait = nda_cultist 
		has_trait = nda_acolyte 
		has_trait = nda_master  
		has_trait = nda_yig_cultist 
		has_trait = nda_yig_acolyte 
		has_trait = nda_yig_master   
	}
}
