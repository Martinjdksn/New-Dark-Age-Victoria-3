nda_nen_newbrunswick_union = {
	is_shown = {
		exists = c:NEN
		this = c:NEN
        exists = c:NBS
	}

	possible = {
		exists = c:NEN
        exists = c:NBS

        # on border
        has_strategic_land_adjacency = c:NBS

		ROOT = { relations:c:NBS >= relations_threshold:neutral }
		NOT = { has_war_with = c:NBS }
		NOT = { owes_obligation_to =  c:NBS }

        # has same government form?
        #custom_tooltip = {
		#	text = nda_nen_newbrunswick_union_gov_tooltip
        #
        #    c:NBS = { has_government_type = c:NEN.government_type }
		#}

	}

    	
	when_taken = {

 		custom_tooltip = nda_nen_newbrunswick_union_tooltip       
        annex = c:NBS

        # add anglo-candian as acceptable
        add_primary_culture = cu:anglo_canadian
	}

	ai_chance = {
		base = 20

		modifier = {
			trigger = { bureaucracy > 0 }
			add = 10
		}
		modifier = {
			OR = {
				is_diplomatic_play_committed_participant = yes
				is_at_war = yes
			}
			add = -1000
		}		
	}

}
