	}

	possible = {

	}
	
	when_taken = {

		# release tibet - needs to be Leng?
		c:TIB = {
			make_independent = yes

			# make cultists
			IF = {
				limit = {
					is_cultist_country = no

				trigger_event = { id = nda_base.7 }		
			}
		}

			limit = { NOT = { exists = c:ELD } }

							this = c:DGN 
							this = c:TIB
						}
					}					
				}

				set_state_owner = c:TIB

				}		
			}

		}

			# rename tibet as Leng and set cultists (or have Leng country?)	- do cultists IG exist at this stage? Or need to wait till nda_base.7 completes?
			ig:ig_nda_cult_worshippers = {

				IF = {
					limit = { NOT = { has_modifier = nda_ig_secret_masters } }
					add_modifier = {
						name = nda_ig_secret_masters
					}
				}
			}

			change_tag = LEN # now change to Leng
		}
		custom_tooltip = nda_dgn_release_leng_tooltip
	}

		base = 10	
	}

}
